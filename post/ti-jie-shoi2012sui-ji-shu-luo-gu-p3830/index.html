代码

#include <cstdio>
int n, op;
double ans, f[105][105];
void solve1() {
	for(int i = 2; i <= n; ++i) ans += 2.0 / i;
}
void solve2() {
	for(int i = 1; i <= n; ++i) f[i][0] = 1;
	for(int i = 2; i <= n; ++i)
		for(int j = 1; j < i; ++j)
			for(int k = 1; k < i; ++k)
				f[i][j] += (f[k][j-1] + f[i-k][j-1] - f[k][j-1]*f[i-k][j-1]) / (i-1);
	for(int i = 1; i < n; ++i) ans += f[n][i];
}
int main() {
	scanf("%d %d", &op, &n);
	if(op & 1) solve1();
	else solve2();
	printf("%.6lf\n", ans);
}
