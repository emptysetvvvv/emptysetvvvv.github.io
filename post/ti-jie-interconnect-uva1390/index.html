题解 Interconnect UVA1390

发布于 2020-08-31 | 9分钟 | 1813字数

背景

小萌新 ∅\varnothing∅ 很委屈，她只是想要小哥哥的赞。

思路 & 代码

题意：给 nnn 个点，已有 mmm 条边，每次随机加边，求使图联通期望步骤数。

我们知道，开始时图上已经有了一些连通分量。显然，想要知道期望值，我们只关心现在有几个连通分量，大小分别是多少。

连通分量的大小用并查集轻松处理，将每个集合的大小（点数）按由小到大的顺序存入 vector vvv 中作为状态。

动态规划的转移用记忆化搜索实现，f(v)f(v)f(v) 表示状态 vvv 的期望步骤数。设 uuu 为连接 vvv 中第 i,ji,ji,j 个连通分量后的状态. 则有：

f(v)=aa−b+∑i,j∈vi≠jvi⋅vja−b⋅f(u)f(v)=\frac{a}{a-b}+\sum_{i,j\in v}^{i\neq j}\frac{v_i\cdot v_j}{a-b}\cdot f(u) f(v)=a−ba​+i,j∈v∑i​=j​a−bvi​⋅vj​​⋅f(u)

其中 aaa 是 nnn 个点所能形成的总边数即 n(n−1)2\dfrac{n(n-1)}{2}2n(n−1)​，bbb 是当前状态已有边数即 ∑i∈vvi(vi−1)2\sum_{i\in v}\dfrac{v_i(v_i -1)}{2}∑i∈v​2vi​(vi​−1)​。

其实蛮好理解的，随机连一条边，可能没起到任何作用。假设当前状态下期望在无用的边上浪费 EEE 次（下一次连到有用边），则 E=1+baEE=1+\dfrac{b}{a}EE=1+ab​E，（本身消耗一次，有 ba\dfrac{b}{a}ab​ 的概率接着磨蹭），得 E=aa−bE=\dfrac{a}{a-b}E=a−ba​。

连接连通分量 i,ji,ji,j 共有 vi⋅vjv_i\cdot v_jvi​⋅vj​ 种方案，故这条有用边连到 i,ji,ji,j 的概率为 vi⋅vja−b\dfrac{v_i\cdot v_j}{a-b}a−bvi​⋅vj​​，贡献为 f(u)f(u)f(u)。

期望值用 map<vector<int>, double> dpdpdp 储存即可.

#include <map>
#include <vector>
#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
int fa[35], size[35], n, m, all;
vector<int> v;
map<vector<int>, double> dp;
int find(int x) {
    return fa[x] == x ? x : fa[x] = find(fa[x]);
}
void merge(int a, int b) {
    int x = find(a), y = find(b);
    if(x != y) fa[y] = x, size[x] += size[y];
}
vector<int> connect(vector<int> v, int a, int b) {
    v.insert(lower_bound(v.begin(), v.end(), v[a] + v[b]), v[a] + v[b]);
    v.erase(v.begin() + a); v.erase(v.begin() + b);
    return v;
}
double f(vector<int> v) {
    if(dp.count(v)) return dp[v];
    int tot = v.size(), edge = 0;
    if(tot == 1) return dp[v] = 0;
    for(int i = 0; i < tot; i++) 
        edge += v[i] * (v[i] - 1) / 2;
    edge = all - edge;
    double ans = (double)all / edge;
    for(int i = 0; i < tot; i++)
        for(int j = 0; j < i; j++)
            ans += (double)v[i] * v[j] / edge * f(connect(v, i, j));
    return dp[v] = ans;
}
int main() {
    cin >> n >> m;
    all = n * (n - 1) / 2;
    for(int i = 1; i <= n; i++) fa[i] = i, size[i] = 1;
    for(int i = 1, a, b; i <= m; i++) 
        cin >> a >> b, merge(a, b);
    for(int i = 1; i <= n; i++) 
        if(fa[i] == i) v.push_back(size[i]);
    sort(v.begin(), v.end());
    printf("%.6lf\n", f(v));
    return 0;
}

事实上，这样的做法在 UVa(luogu) 是可以过的，但是由于 map 和 vector 的大常数，在 poj 会被卡 qwq。

言归正传，如何降复杂度呢？既然只有303030个点，考虑转换成303030进制数完美哈希，对 100007100007100007 取模。

这样的话就很清晰了，用 struct state { double E; bool flag; int v[31]; } 来存状态，长度为303030的 vvv 数组存这个303030进制数的每一位（即每一个连通分量的大小），以哈希值 ∑i∈vvi⋅30i( mod  100007)\sum_{i\in v}v_i\cdot30^i(\bmod\ 100007)∑i∈v​vi​⋅30i(mod 100007) 作为下标存入 state dpdpdp 中，就可以实现了。

你可能需要自己编写一个 hash 函数来获取一个状态的哈希值。由于不能使用 map 了，需要手写 vis 函数实现记忆化的标记，以及 locate 查询某状态在 dpdpdp 数组的下标（没有则返回−1-1−1）。

#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
const int mod = 1e5 + 7;
int n, m, all, fa[32], size[32];
struct state {
    int v[32]; int tot;
    double expect;
    bool flag;
    bool operator == (const state& aim) {
        if(tot != aim.tot) return false;
        for(int i = 0; i < tot; i++)
            if(v[i] != aim.v[i]) return false;
        return true;
    }
    bool operator != (const state& aim) { return !(*this == aim); }
    void clear() {
        memset(v, 0, sizeof v);
        expect = (double)0;
        tot = flag = 0;
    }
    int Hash() {
        int x = 0;
        for(int i = 0; i < tot; i++) x = (x * 30 + v[i]) % mod;
        return x;
    }
} origin, dp[mod + 1];
int locate(state cur) {
    int x = cur.Hash();
    while(dp[x].flag and dp[x] != cur) if(++x == mod) x = 0;
    return dp[x] == cur ? x : -1;
}
void vis(state cur) {
    int x = cur.Hash();
    while(dp[x].flag) if(++x == mod) x = 0;
    dp[x] = cur;
}
state connect(state cur, int i, int j) {
    cur.v[i] = cur.v[i] + cur.v[j], cur.v[j] = 31;
    sort(cur.v, cur.v + cur.tot);
    cur.v[cur.tot--] = 0;
    return cur;
}
int find(int x) {
    return fa[x] == x ? x : fa[x] = find(fa[x]);
}
void merge(int a, int b) {
    int x = find(a), y = find(b);
    if(x != y) fa[x] = y, size[y] += size[x];
}
double f(state cur) {
    if(cur.tot == 1) return (double)0;
    int pos = locate(cur); if(~pos) return dp[pos].expect;
    int now = 0;
    for(int i = 0; i < cur.tot; i++) 
        now += cur.v[i] * (cur.v[i] - 1) / 2;
    now = all - now;
    double ans = (double)all / now;
    for(int i = 0; i < cur.tot; i++)
        for(int j = 0; j < i; j++) {
            double P = (double)cur.v[i] * cur.v[j] / now;
            ans += P * f(connect(cur, i, j));
        }
    cur.flag = true, cur.expect = ans;
    vis(cur);
    return ans;
}
int main() {
    while(~scanf("%d %d", &n, &m)) {
        origin.clear();
        for(int i = 0; i < mod; i++) dp[i].flag = false;
        all = n * (n - 1) / 2;
        for(int i = 1; i <= n; i++) fa[i] = i, size[i] = 1;
        for(int i = 1, u, v; i <= m; i++)
            scanf("%d %d", &u, &v), merge(u, v);
        for(int i = 1; i <= n; i++)
            if(fa[i] == i) origin.v[origin.tot++] = size[i];
        sort(origin.v, origin.v + origin.tot);
        printf("%.6lf\n", f(origin));
    }
    return 0;
}
